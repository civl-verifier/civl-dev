var copy_thread: Tid;
var is_running: [Tid]bool;
var latest: Set<int>;
var copy: Set<int>;
var requested_copy_bound: int;
var copy_bound: int;
var N: int;

Precondition of Main: 
    N == 0 &&
    !is_running[copy_thread] && 
    latest == {} &&
    copy == {} && 
    requested_copy_bound == 0 && 
    copy_bound == 0

Postcondition of Main:
    copy_bound == requested_copy_bound &&
    copy == {n : int :: n in latest && n <= requested_copy_bound}

Inductive invariant Inv:
    copy ⊆ latest ⊆ [1..N]
    copy_bound <= requested_copy_bound <= N
    is_running[copy_thread] ==> latest ⊆ [1..requested_copy_bound]
    !is_running[copy_thread] ==>
        copy == latest ∩ [1..requested_copy_bound] &&
        copy_bound == requested_copy_bound

/// Original program
procedure Main() {
    while (*) {
        call MainLoopBody();
    }
}

action MainLoopBody() {
    async Write(N+1);
    if (*) {
        requested_copy_bound := N;
        if (!is_running[copy_thread]) {
            havoc copy_thread; 
            assume is_running[copy_thread];
            async Copy(copy_thread);
        }
    }
    N := N + 1;
}

action Write(n: int) {
    yield n <= N;
    assume !is_running[copy_thread] || n <= requested_copy_bound;
    if (n <= copy_bound) {
        copy += n;
    }
    latest += n;
}

action Copy(t: Tid) {
    yield t == copy_thread && is_running[copy_thread];
    copy := latest;
    copy_bound := requested_copy_bound;
    is_running[copy_thread] := false;
}

/// Abstract program
procedure Main() {
    while (*) {
        call MainLoopBodyAbs();
    }
}

action MainLoopBodyAbs() {
    assert Inv;
    havoc globals;
    assume Inv;
    if (!old(is_running[copy_thread]) && is_running[copy_thread]) {
        async CopyAbs(*);
    }
}

// left mover
action WriteAbs(n: int) {
    assert Inv;
    havoc globals;
    assume Inv;
    if (!old(is_running[copy_thread]) && is_running[copy_thread]) {
        async CopyAbs(*);
    }
}

// left mover
action CopyAbs(t: Tid) {
    assert Inv;
    havoc globals;
    assume Inv;
    if (!old(is_running[copy_thread]) && is_running[copy_thread]) {
        async CopyAbs(*);
    }
}

/// Inductive sequentialization
action MainLoopBodyAbsInv() {
    assert !is_running[copy_thread];
    assert Inv;
    havoc globals;
    assume Inv;
    if (*) {
        async Write(*);
    }
    while (*) {
        async CopyAbs(*);
    }
    if (is_running[copy_thread]) {
        async CopyAbs(*);
    }
}

action MainLoopBodyAbs'() {
    assert !is_running[copy_thread];
    assert Inv;
    havoc globals;
    assume Inv;
    assume !is_running[copy_thread]);
}
